<project_specification>
  <project_name>EirMed</project_name>

  <overview>
    Sistema pessoal de controle de ficha médica que centraliza todas as informações
    de saúde do usuário em um único lugar. Permite organizar e apresentar rapidamente
    o histórico médico completo ao visitar diferentes profissionais de saúde.
  </overview>

  <technology_stack>
    <backend>
      <framework>ASP.NET Web API (.NET 8)</framework>
      <orm>Entity Framework Core</orm>
      <database>PostgreSQL</database>
      <authentication>ASP.NET Identity + Google SSO</authentication>
    </backend>
    <frontend>
      <framework>Next.js 14+ (App Router)</framework>
      <styling>Tailwind CSS</styling>
      <language>TypeScript</language>
    </frontend>
    <storage>
      <phase1>Local File System</phase1>
      <phase2>Azure Blob Storage</phase2>
    </storage>
    <structure>Monorepo</structure>
  </technology_stack>

  <core_capabilities>
    <capability>Cadastro de Profissionais de Saúde</capability>
    <capability>Registro de Consultas Médicas</capability>
    <capability>Centralização de Exames</capability>
    <capability>Controle de Medicamentos (Farmacinha Pessoal)</capability>
    <capability>Linha do Tempo Médica Unificada</capability>
    <capability>Upload e Armazenamento de Arquivos</capability>
    <capability>Autenticação Multi-usuário com Google SSO</capability>
  </core_capabilities>

  <implemented_features>
    <feature>
      <name>Setup do Monorepo</name>
      <description>Criar estrutura inicial do monorepo com pastas para backend (API) e frontend (web). Configurar workspace e scripts compartilhados.</description>
    </feature>
    <feature>
      <name>Setup do Backend ASP.NET Web API</name>
      <description>Criar projeto ASP.NET Web API com estrutura Clean Architecture. Configurar Entity Framework Core, injeção de dependências e estrutura de pastas.</description>
    </feature>
    <feature>
      <name>Setup do Frontend Next.js</name>
      <description>Criar projeto Next.js 14+ com App Router, TypeScript e Tailwind CSS. Configurar estrutura de componentes e páginas.</description>
    </feature>
    <feature>
      <name>Setup do Banco de Dados PostgreSQL</name>
      <description>Configurar conexão com PostgreSQL, criar DbContext, configurar migrations e criar entidades base do domínio.</description>
    </feature>
    <feature>
      <name>Autenticação com Google SSO</name>
      <description>Implementar autenticação multi-usuário utilizando Google OAuth 2.0 / OpenID Connect. Configurar ASP.NET Identity e JWT para sessões.</description>
    </feature>
    <feature>
      <name>Perfil do Usuário</name>
      <description>CRUD do perfil do usuário com dados pessoais e informações médicas básicas (tipo sanguíneo, alergias, observações gerais).</description>
    </feature>
    <feature>
      <name>CRUD de Profissionais de Saúde</name>
      <description>Cadastro completo de profissionais de saúde (médicos, nutricionistas, fisioterapeutas, etc.) que atendem o usuário.</description>
    </feature>
    <feature>
      <name>CRUD de Consultas Médicas</name>
      <description>Registro detalhado de cada consulta/atendimento realizado, incluindo queixa principal, diagnósticos e observações.</description>
    </feature>
    <feature>
      <name>CRUD de Exames</name>
      <description>Centralização de todos os exames realizados com suporte a diferentes tipos (sangue, imagem, funcional) e anexo de resultados.</description>
    </feature>
    <feature>
      <name>CRUD de Medicamentos (Farmacinha)</name>
      <description>Cadastro de medicamentos disponíveis na farmacinha pessoal com controle de estoque, validade e alertas.</description>
    </feature>
    <feature>
      <name>CRUD de Prescrições</name>
      <description>Registro de prescrições médicas vinculando medicamentos a consultas, com período de uso e médico prescritor.</description>
    </feature>
    <feature>
      <name>Linha do Tempo Médica</name>
      <description>Visualização cronológica unificada de todo o histórico médico: consultas, exames, diagnósticos e medicamentos. Ideal para apresentação rápida durante consultas.</description>
    </feature>
    <feature>
      <name>Upload de Arquivos (Local)</name>
      <description>Implementar upload e download de arquivos (receitas, resultados de exames, relatórios) com armazenamento local. Preparado para migração futura para Azure Blob.</description>
    </feature>
  </implemented_features>
</project_specification>
